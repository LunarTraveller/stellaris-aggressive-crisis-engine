# Basic yes or no, fail or pass indication.
defined_text = {
	name = GetName_acemod_crisis_manager_active
	text = {
		trigger = {
			has_global_flag = acemod_crisis_manager_active
		}
		localization_key = acemod_crisis_manager_trigger_yes
	}
	text = {
		trigger = {
			NOT = {
				has_global_flag = acemod_crisis_manager_active
			}
		}
		localization_key = acemod_crisis_manager_trigger_no
	}
}
